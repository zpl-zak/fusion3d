18 //
19 // Limitations:
20 // - no jpeg progressive support
21 // - non-HDR formats support 8-bit samples only (jpeg, png)
22 // - no delayed line count (jpeg) -- IJG doesn't support either
23 // - no 1-bit BMP
24 // - GIF always returns *comp=4
25 //
26 // Basic usage (see HDR discussion below):
27 // int x,y,n;
28 // unsigned char *data = stbi_load(filename, &x, &y, &n, 0);
29 // // ... process data if not NULL ...
30 // // ... x = width, y = height, n = # 8-bit components per pixel ...
31 // // ... replace '0' with '1'..'4' to force that many components per pixel
32 // // ... but 'n' will always be the number that it would have been if you said 0
33 // stbi_image_free(data)
34 //
35 // Standard parameters:
36 // int *x -- outputs image width in pixels
37 // int *y -- outputs image height in pixels
38 // int *comp -- outputs # of image components in image file
39 // int req_comp -- if non-zero, # of image components requested in result
40 //
41 // The return value from an image loader is an 'unsigned char *' which points
42 // to the pixel data. The pixel data consists of *y scanlines of *x pixels,
43 // with each pixel consisting of N interleaved 8-bit components; the first
44 // pixel pointed to is top-left-most in the image. There is no padding between
45 // image scanlines or between pixels, regardless of format. The number of
46 // components N is 'req_comp' if req_comp is non-zero, or *comp otherwise.
47 // If req_comp is non-zero, *comp has the number of components that _would_
48 // have been output otherwise. E.g. if you set req_comp to 4, you will always
49 // get RGBA output, but you can check *comp to easily see if it's opaque.
50 //
51 // An output image with N components has the following components interleaved
52 // in this order in each pixel:
53 //
54 // N=#comp components
55 // 1 grey
56 // 2 grey, alpha
57 // 3 red, green, blue
58 // 4 red, green, blue, alpha
59 //
60 // If image loading fails for any reason, the return value will be NULL,
61 // and *x, *y, *comp will be unchanged. The function stbi_failure_reason()
62 // can be queried for an extremely brief, end-user unfriendly explanation
63 // of why the load failed. Define STBI_NO_FAILURE_STRINGS to avoid
64 // compiling these strings at all, and STBI_FAILURE_USERMSG to get slightly
65 // more user-friendly ones.
66 //
67 // Paletted PNG, BMP, GIF, and PIC images are automatically depalettized.
68 //
69 // ===========================================================================
70 //
71 // iPhone PNG support:
72 //
73 // By default we convert iphone-formatted PNGs back to RGB; nominally they
74 // would silently load as BGR, except the existing code should have just
75 // failed on such iPhone PNGs. But you can disable this conversion by
76 // by calling stbi_convert_iphone_png_to_rgb(0), in which case
77 // you will always just get the native iphone "format" through.
78 //
79 // Call stbi_set_unpremultiply_on_load(1) as well to force a divide per
80 // pixel to remove any premultiplied alpha *only* if the image file explicitly
81 // says there's premultiplied data (currently only happens in iPhone images,
82 // and only if iPhone convert-to-rgb processing is on).
83 //
84 // ===========================================================================
85 //
86 // HDR image support (disable by defining STBI_NO_HDR)
87 //
88 // stb_image now supports loading HDR images in general, and currently
89 // the Radiance .HDR file format, although the support is provided
90 // generically. You can still load any file through the existing interface;
91 // if you attempt to load an HDR file, it will be automatically remapped to
92 // LDR, assuming gamma 2.2 and an arbitrary scale factor defaulting to 1;
93 // both of these constants can be reconfigured through this interface:
94 //
95 // stbi_hdr_to_ldr_gamma(2.2f);
96 // stbi_hdr_to_ldr_scale(1.0f);
97 //
98 // (note, do not use _inverse_ constants; stbi_image will invert them
99 // appropriately).
100 //
101 // Additionally, there is a new, parallel interface for loading files as
102 // (linear) floats to preserve the full dynamic range:
103 //
104 // float *data = stbi_loadf(filename, &x, &y, &n, 0);
105 //
106 // If you load LDR images through this interface, those images will
107 // be promoted to floating point values, run through the inverse of
108 // constants corresponding to the above:
109 //
110 // stbi_ldr_to_hdr_scale(1.0f);
111 // stbi_ldr_to_hdr_gamma(2.2f);
112 //
113 // Finally, given a filename (or an open file or memory block--see header
114 // file for details) containing image data, you can query for the "most
115 // appropriate" interface to use (that is, whether the image is HDR or
116 // not), using:
117 //
118 // stbi_is_hdr(char *filename);
119 //
120 // ===========================================================================
121 //
122 // I/O callbacks
123 //
124 // I/O callbacks allow you to read from arbitrary sources, like packaged
125 // files or some other source. Data read from callbacks are processed
126 // through a small internal buffer (currently 128 bytes) to try to reduce
127 // overhead.
128 //
129 // The three functions you must define are "read" (reads some bytes of data),
130 // "skip" (skips some bytes of data), "eof" (reports if the stream is at the end).
